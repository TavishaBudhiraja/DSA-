#include <stdio.h>
int main() {
    int arr[] = {1, 0, -2, 3, -5, 0, 7};
    int n = sizeof(arr) / sizeof(arr[0]);
    int k = 0; 
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            int temp = arr[i];   
            int j = i;
            while (j > k) {
                arr[j] = arr[j - 1];
                j--;
            }
            arr[k] = temp;      
            k++;                
        }
    }
    printf("Rearranged array: ");
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");
    return 0;
}
