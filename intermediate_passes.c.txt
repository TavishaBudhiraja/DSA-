#include <stdio.h>
int main() {
    int n, i, j, temp;
    printf("Enter size: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter elements: ");
    for(i=0; i<n; i++) {
        scanf("%d", &arr[i]);
    }
    for(i=0; i<n-1; i++) {
        for(j=0; j<n-1-i; j++) {
            if(arr[j] > arr[j+1]) {
                temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
        printf("Pass %d: ", i+1);
        for(j=0; j<n; j++) {
            printf("%d ", arr[j]);
        }
        printf("\n");
    }
    return 0;
}
