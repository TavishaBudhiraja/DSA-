#include <stdio.h>
int main() {
    int n, i, j, temp;
    printf("Enter size: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements: ", n);
    for(i=0; i<n; i++) {
        scanf("%d", &arr[i]);
    }
    for(i=0; i<n-1; i++) {
        for(j=i+1; j<n; j++) {
            if(arr[i] < arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    if(n >= 3)
        printf("Second largest = %d\nThird largest = %d\n", arr[1], arr[2]);
    else
        printf("Not enough numbers!\n");
    return 0;
}
